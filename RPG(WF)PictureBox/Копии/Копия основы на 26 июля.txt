        public void Если_враг(int z) // Если_враг
        {
            DialogResult результат = new DialogResult();
            Окно_боя окно = new Окно_боя();
            int y = вар[o, r].Атака;
            окно.заполнение(хар.Атака, y);
            результат = окно.ShowDialog();
            if (результат == DialogResult.No)
            {
                z = 0;
                return;
            }
            if (результат == DialogResult.Yes)
            {
                Бой бой = new Бой(хар.Атака, вар[o, r].Атака);
                if (бой.результат == 2)
                {
                    MessageBox.Show("Вы погибли");
                    (panel3.Controls["pctdd"] as PictureBox).Dispose();
                    z = 0;
                    return;
                }
                if (бой.результат == 1)
                {
                    z = 1;
                }
            }
        }
